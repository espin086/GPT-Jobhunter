133class KeywordMatcher: 

134 """A matcher for keywords. 

135 

136 Given a list of names, matches any substring of one of these names. The 

137 string inclusion check is case-insensitive. 

138 

139 Will match on the name of colitem, including the names of its parents. 

140 Only matches names of items which are either a :class:`Class` or a 

141 :class:`Function`. 

142 

143 Additionally, matches on names in the 'extra_keyword_matches' set of 

144 any item, as well as names directly assigned to test functions. 

145 """ 

146 

147 __slots__ = ("_names",) 

148 

149 _names: AbstractSet[str] 

150 

151 @classmethod 

152 def from_item(cls, item: "Item") -> "KeywordMatcher": 

153 mapped_names = set() 

154 

155 # Add the names of the current item and any parent items. 

156 import pytest 

157 

158 for node in item.listchain(): 

159 if not isinstance(node, pytest.Session): 

160 mapped_names.add(node.name) 

161 

162 # Add the names added as extra keywords to current or parent items. 

163 mapped_names.update(item.listextrakeywords()) 

164 

165 # Add the names attached to the current function through direct assignment. 

166 function_obj = getattr(item, "function", None) 

167 if function_obj: 

168 mapped_names.update(function_obj.__dict__) 

169 

170 # Add the markers to the keywords as we no longer handle them correctly. 

171 mapped_names.update(mark.name for mark in item.iter_markers()) 

172 

173 return cls(mapped_names) 

174 

175 def __call__(self, subname: str) -> bool: 

176 subname = subname.lower() 

177 names = (name.lower() for name in self._names) 

178 

179 for name in names: 

180 if subname in name: 

181 return True 

182 return False

185def deselect_by_keyword(items: "List[Item]", config: Config) -> None: 

186 keywordexpr = config.option.keyword.lstrip() 

187 if not keywordexpr: 

188 return 

189 

190 expr = _parse_expression(keywordexpr, "Wrong expression passed to '-k'") 

191 

192 remaining = [] 

193 deselected = [] 

194 for colitem in items: 

195 if not expr.evaluate(KeywordMatcher.from_item(colitem)): 

196 deselected.append(colitem) 

197 else: 

198 remaining.append(colitem) 

199 

200 if deselected: 

201 config.hook.pytest_deselected(items=deselected) 

202 items[:] = remaining
