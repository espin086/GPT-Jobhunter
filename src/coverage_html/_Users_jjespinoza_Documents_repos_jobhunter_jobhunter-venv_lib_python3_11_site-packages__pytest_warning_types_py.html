148def warn_explicit_for(method: FunctionType, message: PytestWarning) -> None: 

149 """ 

150 Issue the warning :param:`message` for the definition of the given :param:`method` 

151 

152 this helps to log warnings for functions defined prior to finding an issue with them 

153 (like hook wrappers being marked in a legacy mechanism) 

154 """ 

155 lineno = method.__code__.co_firstlineno 

156 filename = inspect.getfile(method) 

157 module = method.__module__ 

158 mod_globals = method.__globals__ 

159 try: 

160 warnings.warn_explicit( 

161 message, 

162 type(message), 

163 filename=filename, 

164 module=module, 

165 registry=mod_globals.setdefault("__warningregistry__", {}), 

166 lineno=lineno, 

167 ) 

168 except Warning as w: 

169 # If warnings are errors (e.g. -Werror), location information gets lost, so we add it to the message. 

170 raise type(w)(f"{w}\n at {filename}:{lineno}") from None
