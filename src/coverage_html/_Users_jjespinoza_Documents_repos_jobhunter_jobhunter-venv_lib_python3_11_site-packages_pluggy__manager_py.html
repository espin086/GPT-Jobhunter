117 def register(self, plugin: _Plugin, name: str | None = None) -> str | None: 

118 """Register a plugin and return its name. 

119 

120 :param name: 

121 The name under which to register the plugin. If not specified, a 

122 name is generated using :func:`get_canonical_name`. 

123 

124 :returns: 

125 The plugin name. If the name is blocked from registering, returns 

126 ``None``. 

127 

128 If the plugin is already registered, raises a :exc:`ValueError`. 

129 """ 

130 plugin_name = name or self.get_canonical_name(plugin) 

131 

132 if plugin_name in self._name2plugin: 

133 if self._name2plugin.get(plugin_name, -1) is None: 

134 return None # blocked plugin, return None to indicate no registration 

135 raise ValueError( 

136 "Plugin name already registered: %s=%s\n%s" 

137 % (plugin_name, plugin, self._name2plugin) 

138 ) 

139 

140 if plugin in self._name2plugin.values(): 

141 raise ValueError( 

142 "Plugin already registered under a different name: %s=%s\n%s" 

143 % (plugin_name, plugin, self._name2plugin) 

144 ) 

145 

146 # XXX if an error happens we should make sure no state has been 

147 # changed at point of return 

148 self._name2plugin[plugin_name] = plugin 

149 

150 # register matching hook implementations of the plugin 

151 for name in dir(plugin): 

152 hookimpl_opts = self.parse_hookimpl_opts(plugin, name) 

153 if hookimpl_opts is not None: 

154 normalize_hookimpl_opts(hookimpl_opts) 

155 method: _HookImplFunction[object] = getattr(plugin, name) 

156 hookimpl = HookImpl(plugin, plugin_name, method, hookimpl_opts) 

157 name = hookimpl_opts.get("specname") or name 

158 hook: HookCaller | None = getattr(self.hook, name, None) 

159 if hook is None: 

160 hook = HookCaller(name, self._hookexec) 

161 setattr(self.hook, name, hook) 

162 elif hook.has_spec(): 

163 self._verify_hook(hook, hookimpl) 

164 hook._maybe_apply_history(hookimpl) 

165 hook._add_hookimpl(hookimpl) 

166 return plugin_name
