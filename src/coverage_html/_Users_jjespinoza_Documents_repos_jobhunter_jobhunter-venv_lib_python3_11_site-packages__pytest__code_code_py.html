919 def _truncate_recursive_traceback( 

920 self, traceback: Traceback 

921 ) -> Tuple[Traceback, Optional[str]]: 

922 """Truncate the given recursive traceback trying to find the starting 

923 point of the recursion. 

924 

925 The detection is done by going through each traceback entry and 

926 finding the point in which the locals of the frame are equal to the 

927 locals of a previous frame (see ``recursionindex()``). 

928 

929 Handle the situation where the recursion process might raise an 

930 exception (for example comparing numpy arrays using equality raises a 

931 TypeError), in which case we do our best to warn the user of the 

932 error and show a limited traceback. 

933 """ 

934 try: 

935 recursionindex = traceback.recursionindex() 

936 except Exception as e: 

937 max_frames = 10 

938 extraline: Optional[str] = ( 

939 "!!! Recursion error detected, but an error occurred locating the origin of recursion.\n" 

940 " The following exception happened when comparing locals in the stack frame:\n" 

941 " {exc_type}: {exc_msg}\n" 

942 " Displaying first and last {max_frames} stack frames out of {total}." 

943 ).format( 

944 exc_type=type(e).__name__, 

945 exc_msg=str(e), 

946 max_frames=max_frames, 

947 total=len(traceback), 

948 ) 

949 # Type ignored because adding two instances of a List subtype 

950 # currently incorrectly has type List instead of the subtype. 

951 traceback = traceback[:max_frames] + traceback[-max_frames:] # type: ignore 

952 else: 

953 if recursionindex is not None: 

954 extraline = "!!! Recursion detected (same locals & position)" 

955 traceback = traceback[: recursionindex + 1] 

956 else: 

957 extraline = None 

958 

959 return traceback, extraline

961 def repr_excinfo( 

962 self, excinfo: ExceptionInfo[BaseException] 

963 ) -> "ExceptionChainRepr":

976 # full support for exception groups added to ExceptionInfo. 

977 # See https://github.com/pytest-dev/pytest/issues/9159

984 excinfo_.value, 

985 excinfo_.traceback[0]._rawentry, 

986 ) 

987 ) 

988 else: 

989 reprtraceback = self.repr_traceback(excinfo_) 

990 reprcrash = excinfo_._getreprcrash() 

991 else: 

992 # Fallback to native repr if the exception doesn't have a traceback: 

993 # ExceptionInfo objects require a full traceback to work. 

994 reprtraceback = ReprTracebackNative( 

995 traceback.format_exception(type(e), e, None) 

996 ) 

997 reprcrash = None 

998 repr_chain += [(reprtraceback, reprcrash, descr)] 

999

1003 descr = "The above exception was the direct cause of the following exception:"

1006 ): 

1007 e = e.__context__ 

1008 excinfo_ = ExceptionInfo.from_exception(e) if e.__traceback__ else None 

1009 descr = "During handling of the above exception, another exception occurred:" 

1010 else: 

1011 e = None 

1012 repr_chain.reverse()

1141 def _write_entry_lines(self, tw: TerminalWriter) -> None: 

1142 """Write the source code portions of a list of traceback entries with syntax highlighting. 

1143 

1144 Usually entries are lines like these: 

1145 

1146 " x = 1" 

1147 "> assert x == 2" 

1148 "E assert 1 == 2" 

1149

1151 the "E" prefix) using syntax highlighting, taking care to not highlighting the ">" 

1152 character, as doing so might break line continuations. 

1153 """ 

1154 

1155 if not self.lines: 

1156 return 

1157

1159 # highlight the code but not the indentation, which may contain markers 

1160 # such as "> assert 0"

1162 indent_size = len(fail_marker)

1167 is_failure_line = line.startswith(fail_marker) 

1168 if is_failure_line: 

1169 # from this point on all lines are considered part of the failure 

1170 failure_lines.extend(self.lines[index:]) 

1171 break 

1172 else: 

1173 if self.style == "value":

1175 else:

1177 source_lines.append(line[indent_size:])

1182 for line in failure_lines:
